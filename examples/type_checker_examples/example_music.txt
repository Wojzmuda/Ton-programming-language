$ 1. Test funkcji zwracających dźwięk z argumentami
!define SOUND stworz_glowny_dzwiek <INT czas_trwania> {
    !make SOUND n1 <- synth C4 czas_trwania;
    
    $ Tymczasowo zwracamy pojedynczy dźwięk (omijamy operator +)
    $ Type Checker poprawnie zweryfikuje, że zwracamy typ SOUND
    !out n1; 
}

!define SOUND stworz_szybki_dzwiek <INT czas_trwania> {
    !make SOUND n1 <- synth G4 czas_trwania;
    
    $ Tymczasowo zwracamy pojedynczy dźwięk (omijamy operator &)
    !out n1; 
}

$ 2. Inicjalizacja głównej osi czasu i ścieżek 
!make TIMELINE cyber_track;
cyber_track NEW TRACK akordy;
cyber_track NEW TRACK melodia;

$ 3. Wywołanie funkcji (Type Checker weryfikuje argumenty INT i typ zwracany)
!make SOUND glowny_akord <- stworz_glowny_dzwiek(1000);
!make SOUND szybki_arp <- stworz_szybki_dzwiek(250);

$ 4. Operacje na tablicach zdarzeń (Event Lists) i przypisanie do osi czasu 
$ Ciekawostka: Akordy i miksowanie i tak zadziałają, jeśli ułożysz dźwięki
$ na tej samej milisekundzie za pomocą AT!
cyber_track.akordy <- [glowny_akord AT 0, glowny_akord AT 2000];
cyber_track.melodia <- [szybki_arp AS "intro" AT 500, szybki_arp AT 2500];

$ 5. Manipulacja czasem na konkretnym aliasie 
SHIFT cyber_track.melodia."intro" BY 100;

$ 6. Zapisanie gotowego utworu 
!save cyber_track "cyber_track.wav";
