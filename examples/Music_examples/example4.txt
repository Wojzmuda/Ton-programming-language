!make SOUND x <- piano C#4 1000;
!make SOUND y <- piano D#4 1000;
!make SOUND z <- piano E#4 1000;
!make TIMELINE t1;
t1 NEW TRACK pianos;
t1.pianos <- [x AT 0, y AT 1000, z AT 2000];

!save t1 "a.wav";
